\frametitle{2D Navier--Stokes equation}

\vspace{-4ex}
\begin{equation*}
    \begin{aligned}
        \partial_t \omega(x,t) + v(x,t) \cdot \nabla \omega(x,t) & = \nu \Delta \omega(x,t) + f(x), & \quad x \in \LRp{0,1}^2, t \in (0, T], \\
        \nabla \cdot v(x,t)                                      & = 0,                             & \quad x \in \LRp{0,1}^2, t \in (0, T], \\
        \omega(x,0)                                              & = u(x),                   & \quad x \in \LRp{0,1}^2,
    \end{aligned}
\end{equation*}
